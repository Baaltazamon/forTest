private Tuple<List<Users>, List<Roles>> GetUsersUP(int pageNumber, int pageSize, string role)
        {
            var role_ = db.Roles.SingleOrDefault(c => c.Name.Equals(role));
            if (role_ is null)
                return Tuple.Create<List<Users>, List<Roles>>(new List<Users>(), new List<Roles>());
            var users = Queryable.Where(db.Users, c => Enumerable.Any<Roles>(c.Roles, g => g.Id == role_.Id)).ToList();
            var roles = new List<Roles>();
            foreach (var item in users)
            {
                roles.Add(Queryable.SingleOrDefault(db.Roles, c=> Enumerable.Any<Users>(c.Users, g => g.Id == item.Id)));
            }
            if (pageSize > users.Count)
                return Tuple.Create<List<Users>, List<Roles>>(users, roles);
            var usersPage = new List<Users>();
            var rolesPage = new List<Roles>();
            for (int i = (pageNumber - 1) * pageSize; i < pageSize * pageNumber; i++)
            {
                usersPage.Add(users[i]);
                var rolesS = Queryable
                    .Where(db.Roles, c => Enumerable.Any<Users>(c.Users, g => g.Id == users[i].Id)).ToList();
                foreach (var item in rolesS)
                {
                    rolesPage.Add(item);
                }
            }
            return Tuple.Create<List<Users>, List<Roles>>(usersPage, rolesPage); 
        }

private List<Employees> GetEmployees(string name)
        {
            if (name is null)
                return new List<Employees>();
            Employees employee = Queryable.SingleOrDefault(db.Employees, c => c.Name.Equals(name));
            if (employee is null)
                return new List<Employees>();
            var EmpList = new List<Employees>();
            foreach (var item in db.Employees)
                if (item.ManagerId == employee.Id)
                {
                    EmpList.Add(item);
                    foreach (var item2lvl in db.Employees)
                        if (item2lvl.ManagerId == item.Id)
                            EmpList.Add(item2lvl);
                }

            return EmpList;
        }
